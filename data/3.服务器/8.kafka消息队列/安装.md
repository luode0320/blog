# 拉取

```sh
docker pull luode0320/kafka:latest
docker pull luode0320/kafka-ui:latest
```

# kafka.yaml

```sh
mkdir -p /usr/local/src/kafka
cd /usr/local/src/kafka
vi kafka.yaml
```

```yaml
version: '3'
services:
  kafka:
    image: luode0320/kafka:latest
    container_name: kafka
    environment:
      - KAFKA_CFG_NODE_ID=1 # 节点的 ID，这个 ID 在 Kafka 集群中必须是唯一的
      - KAFKA_CFG_PROCESS_ROLES=controller,broker # 节点既是控制器又是代理，负责集群的控制逻辑又负责消息的存储和转发
      - KAFKA_CFG_ADVERTISED_LISTENERS=PLAINTEXT://192.168.1.16:9092 # 外网IP的 9092 端口
      - KAFKA_CFG_CONTROLLER_LISTENER_NAMES=CONTROLLER # 控制器监听器的名称，这里设置为 CONTROLLER
      - KAFKA_CFG_LISTENERS=PLAINTEXT://0.0.0.0:9092,CONTROLLER://0.0.0.0:9093 # 两个监听器
      - KAFKA_CFG_LISTENER_SECURITY_PROTOCOL_MAP=CONTROLLER:PLAINTEXT,PLAINTEXT:PLAINTEXT # 两个监听器使用明文传输协议
      - KAFKA_CFG_CONTROLLER_QUORUM_VOTERS=1@0.0.0.0:9093 # 选举投票者，使用 0.0.0.0 的 9093 端口
      - KAFKA_KRAFT_CLUSTER_ID=abcdefghijklmnopqrstuv # Kafka Kraft 集群的 ID，用于标识属于同一个 Kraft 集群的 Kafka 节点
      - KAFKA_CFG_LOG_RETENTION_BYTES=524288000 # 设置日志保留的字节数，这里设置为 500M=524288000 1GB=1073741824 5GB=5368709120
    volumes:
      - /usr/local/src/kafka/data:/bitnami/kafka/data
    ports:
      - 9092:9092
      - 9093:9093
    restart: always
  kafka-ui:
    container_name: kafka-ui
    image: luode0320/kafka-ui:latest
    depends_on:
      - kafka
    environment:
      - DEFAULT_USERNAME=admin
      - DEFAULT_PASSWORD=admin
    ports:
      - 9091:8080
```

# 启动

```sh
docker-compose -f /usr/local/src/kafka/kafka.yaml up -d
```
